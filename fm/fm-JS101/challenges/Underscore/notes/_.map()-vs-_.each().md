# _.map() vs _.each()

```javascript
function AnimalMaker(name) {
  return {
    speak: function() {
      console.log("My name is ", name);
    }
  }
}

var animalNames = ["Smokey", "Fluffy", "Trigger"];

var farm = _.map(animalNames, AnimalMaker); // returns an array


_.each(farm, function(animal){
  animal.speak();
});

```
